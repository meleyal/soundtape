Activate a tab wen one is clicked. Passes the selected tab onto the TabView.

 
  activateTab: (e) =>
    e.preventDefault()
    el = $(e.currentTarget)
    action = el.attr('data-action')
    return if el.is('.active')
    @$('nav a').removeClass('active')
    el.addClass 'active'
    app.tabsView.switchTo action
